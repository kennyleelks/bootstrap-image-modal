Bootstrap Image Modal Demo

Events

$("#demo1").imageModal({
    onShown: function() {
        console.log("modal is shown!");
    },
    onHidden: function() {
        console.log("modal is hidden!");
    }
});

Remove animation

$("#demo2").imageModal({
    animate: false
});

Optional sizes

$("#demo3").imageModal({
    modalSize: "lg"
});

$("#demo4").imageModal({
    modalSize: "sm"
});
